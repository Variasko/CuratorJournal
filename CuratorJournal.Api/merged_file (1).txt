﻿namespace ConsoleApp1
{
    public class StudyGroup
    {
        public int GroupId { get; set; }
        public required string DirectionAbbreviation { get; set; }
        public string? SpecializationAbbreviation { get; set; }
        public int Course { get; set; }
        public DateOnly DateCreated { get; set; }
        public bool IsBudget { get; set; }
        public bool IsStudyCompleted { get; set; }
        public required List<Student> Students { get; set; }

        public string FullName
        {
            get
            {
                string year = $"{DateCreated:yy}";
                string specialization = SpecializationAbbreviation ?? string.Empty;
                string fundingSuffix = IsBudget ? string.Empty : "в";

                return $"{DirectionAbbreviation}-{Course}{year}{specialization}{fundingSuffix}";
            }
        }
    }
}
